POST users/
PATCH users/id
DELETE users/id

GET instructors/
GET instructors/id
POST instructors/
PATCH instructors/id

GET lessons/
GET lessons/id
POST lessons/
PATCH lessons/id
DELETE lessons/id

GET /token
POST /token
DELETE /token
